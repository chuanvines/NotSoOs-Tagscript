import json

base = "/home/notsocoder/notsoos/"
fs_path = base + "system/fs.json"

if os.path.exists(fs_path):
    with open(fs_path) as f:
        fs = json.load(f)

    fs = {"total_size": 0, "max_size": 1500000000, "files": {}}

files_info = []
for name, data in fs.get("files", {}).items():
    size_mb = data.get("size", 0) / 1000000
    files_info.append(f"📄 **{name}** - {size_mb:.2f}MB")

files_display = "\n".join(files_info) if files_info else "No files stored"

used = fs.get("total_size", 0)
max_size = fs.get("max_size", 1500000000)
used_mb = used / 1000000
max_mb = max_size / 1000000
free_mb = (max_size - used) / 1000000
usage_percent = (used / max_size * 100) if max_size > 0 else 0

print(json.dumps({
    'files': files_display,
    'total': len(fs.get('files', {})),
    'used': f'{used_mb:.2f}MB',
    'max': f'{max_mb:.2f}MB',
    'free': f'{free_mb:.2f}MB',
    'usage': f'{usage_percent:.1f}%'
}, separators=(',', ':')))
}
