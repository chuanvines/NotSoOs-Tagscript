{python:
import os
import json

base = "/home/notsocoder/notsoos/"

def save_file(name, content):
    with open(base + "system/fs.json") as f:
        fs = json.load(f)
    
    content_bytes = len(content.encode())
    if fs["total_size"] + content_bytes > fs["max_size"]:
        print("ERROR: Storage full!")
        return
    
    filepath = base + "files/" + name
    with open(filepath, "w") as f:
        f.write(content)
    
    fs["files"][name] = {"size": content_bytes, "path": filepath}
    fs["total_size"] += content_bytes
    
    with open(base + "system/fs.json", "w") as f:
        json.dump(fs, f)
    print(f"Saved {name}")

save_file("readme.txt", "Welcome to NotSoOs!")
}
