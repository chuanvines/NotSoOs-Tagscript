import json

if not os.path.exists(fs_path):
    fs = {"total_size": 0, "max_size": 1500000000, "files": {}}
    with open(fs_path, "w") as f:
        json.dump(fs, f, indent=2)
else:
    with open(fs_path) as f:
        fs = json.load(f)

files_list = "\n".join([f"- {k} ({v['size']} bytes)" for k,v in fs.get("files", {}).items()])
if not files_list:
    files_list = "No files yet"

used = fs.get("total_size", 0)
max_s = 1500000000
free = max_s - used

print(json.dumps({'files': files_list, 'storage': f'{used/1000000:.2f}MB / {max_s/1000000:.2f}MB', 'free': f'{free/1000000:.2f}MB', 'count': len(fs.get('files', {}))}, separators=(',', ':')))
}
